Toggle Button Animation

To see the code click to toggle.

HTML

// Position with Bootstrap how you'd like
<div class="row">
    <a class="canvas-toggle" href="#">
        <div class="menu">
            <div class="top-bar"></div>
            <div class="middle-bar"></div>
            <div class="bottom-bar"></div>
        </div>
    </a>
</div>

CSS

.canvas-toggle .menu {
  display: block;
}

.exit-off-canvas{
  box-shadow: none;
}

.top-bar, .middle-bar, .bottom-bar {
  display:block;
  width: 20px;
  height: 3px;
  background-color: #333;
  margin-bottom: 4px;
  transition: transform 1s
  cubic-bezier(1, 0, 0, 1) 0,
  background-color 0.6s ease 0;
}

.middle-bar{
  transition-property: opacity;
  transition-duration: 1s;
}

.top-bar, .bottom-bar {
  transform-origin: 100% 50%;
}

.top-bar-close{
  transform: rotate(-45deg);
  background-color: grey;
}

.middle-bar-close{
  opacity: 0;
  transition-property:opacity;
  transition-duration: 0.8s;
}

.bottom-bar-close{
  transform: rotate(45deg);
  background-color: grey;
}

Javascript

<script type="text/javascript">
$(function(){
	$(".canvas-toggle, .exit-off-canvas").click(function(){
	$(".top-bar").toggleClass("top-bar-close");
	$(".middle-bar").toggleClass("middle-bar-close");		
	$(".bottom-bar").toggleClass("bottom-bar-close");
	});
});
</script>
